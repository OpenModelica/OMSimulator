#CAPTION#
setLoggingLevel
---------------

Enables/Disables debug logging (logDebug and logTrace).

0 default, 1 default+debug, 2 default+debug+trace
#END#

#LUA#
.. code-block:: lua

  oms_setLoggingLevel(logLevel)

#END#

#PYTHON#
.. code-block:: python

  oms.setLoggingLevel(logLevel)

#END#

#CAPI#
.. code-block:: c

  void oms_setLoggingLevel(int logLevel);

#END#

#END#
