#CAPTION#
freeMemory
----------

Free the memory allocated by some other API. Pass the object for which memory
is allocated.
#END#

#LUA#
.. code-block:: lua

  oms_freeMemory(obj)

#END#

#PYTHON#
.. code-block:: python

  oms.freeMemory(obj)

#END#

#CAPI#
.. code-block:: c

  void oms_freeMemory(void* obj);

#END#
